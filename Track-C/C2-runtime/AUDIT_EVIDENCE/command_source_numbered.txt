    11	struct SubmitState {
    12	    std::mutex mutex;
    13	    uint64_t next_sequence = 1;
    14	    bool exhausted = false;
    15	};
    16	
    17	SubmitState &submit_state() {
    18	    static SubmitState state;
    19	    return state;
    20	}
    21	
    22	} // namespace
    23	
    24	aecError_t submit_and_validate_completion(
    25	    aecDeviceCommand &command, aecDeviceCompletion *completion_out) {
    26	    auto &state = submit_state();
    27	    std::lock_guard<std::mutex> submit_lock(state.mutex);
    28	    if (state.exhausted || state.next_sequence == 0) {
    29	        return AEC_ERROR_INTERNAL;
    30	    }
    31	
    32	    // Sequence assignment and the blocking submit share this lock. A worker
    33	    // can therefore never deliver a larger sequence before a smaller one.
    34	    command.sequence = state.next_sequence;
    35	    if (state.next_sequence == std::numeric_limits<uint64_t>::max()) {
    36	        state.exhausted = true;
    37	    } else {
    38	        ++state.next_sequence;
    39	    }
    40	
    41	    aecDeviceCompletion completion{};
    42	    const aecDeviceStatus submit_status = aecDeviceSubmit(&command, &completion);
    43	    if (completion.abi_version != AEC_DEVICE_ABI_VERSION ||
    44	        completion.sequence != command.sequence) {
    45	        return AEC_ERROR_INTERNAL;
    46	    }
    47	    if (static_cast<uint32_t>(submit_status) != completion.status) {
    48	        return AEC_ERROR_INTERNAL;
    49	    }
    50	    if (completion_out != nullptr) {
    51	        *completion_out = completion;
    52	    }
    53	    return map_device_status(submit_status);
    54	}

    56	aecError_t submit_dma(aecDevicePtr device_ptr, void *host_ptr, uint64_t bytes,
    57	                      aecCopyDirection direction, uint64_t stream_id,
    58	                      uint8_t channel, uint16_t flags) {
    59	    if (host_ptr == nullptr || bytes == 0 || channel > 1) {
    60	        return AEC_ERROR_INVALID_ARGUMENT;
    61	    }
    62	    if (direction != AEC_COPY_HOST_TO_DEVICE &&
    63	        direction != AEC_COPY_DEVICE_TO_HOST) {
    64	        return AEC_ERROR_INVALID_ARGUMENT;
    65	    }
    66	    if ((flags & AEC_DEVICE_FLAG_ZERO_COPY) != 0 &&
    67	        (flags & AEC_DEVICE_FLAG_REGISTERED) == 0) {
    68	        return AEC_ERROR_INVALID_ARGUMENT;
    69	    }
    70	
    71	    aecDeviceCommand command{};
    72	    command.abi_version = AEC_DEVICE_ABI_VERSION;
    73	    command.opcode = direction == AEC_COPY_HOST_TO_DEVICE
    74	                         ? AEC_DEVICE_OP_H2D
    75	                         : AEC_DEVICE_OP_D2H;
    76	    command.flags = flags;
    77	    command.stream_id = stream_id;
    78	    if (direction == AEC_COPY_HOST_TO_DEVICE) {
    79	        command.dst = device_ptr;
    80	    } else {
    81	        command.src = device_ptr;
    82	    }
    83	    command.host_address = static_cast<uint64_t>(
    84	        reinterpret_cast<uintptr_t>(host_ptr));
    85	    command.bytes = bytes;
    86	    command.chunk_bytes = 65536;
    87	    command.queue_depth = 2;
    88	    command.channel = channel;
    89	    return submit_and_validate_completion(command);
    90	}
